#======start of new build.yml, dont copy this line================
name: "Build and Release Tauri App"

on:
  push:
    branches: [ main ]
  workflow_dispatch: 

jobs:
  build:
    strategy:
      fail-fast: false
      matrix:
        platform: [ubuntu-latest, windows-latest]

    runs-on: ${{ matrix.platform }}
    steps:
      - name: Checkout repository
        uses: actions/checkout@v4

      - name: Setup Node.js
        uses: actions/setup-node@v4
        with:
          node-version: 20

      - name: Install Rust stable
        uses: dtolnay/rust-toolchain@stable

      # UPDATED: Using libwebkit2gtk-4.1-dev and libayatana-appindicator3-dev for Ubuntu 24.04 / Tauri v2
      - name: Install Linux dependencies
        if: matrix.platform == 'ubuntu-latest'
        run: |
          sudo apt-get update
          sudo apt-get install -y libwebkit2gtk-4.1-dev libayatana-appindicator3-dev librsvg2-dev patchelf

      # UPDATED: Using the correct Tauri v2 CLI flags (--frontend-dist and --dev-url)
      - name: Generate Tauri Boilerplate
        shell: bash
        run: |
          npm init -y
          npm install @tauri-apps/cli
          npm pkg set scripts.tauri="tauri"
          npx tauri init --app-name robot-shooting-game --window-title "Robot Shooting Game" --frontend-dist "../src" --dev-url "http://localhost:1420"
          cp tauri.conf.json src-tauri/tauri.conf.json

      - name: Set Windows Rust Flags
        if: matrix.platform == 'windows-latest'
        run: echo "RUSTFLAGS=-C target-feature=+crt-static" >> $GITHUB_ENV

      - name: Build and Package Tauri App
        uses: tauri-apps/tauri-action@v0
        env:
          GITHUB_TOKEN: ${{ secrets.GITHUB_TOKEN }}
        with:
          tagName: app-v0.1.0
          releaseName: 'Robot Shooting Game v0.1.0'
          releaseBody: 'Automated build generated via GitHub Actions.'
          releaseDraft: true
          prerelease: false
#======end of new build.yml, dont copy this line================
